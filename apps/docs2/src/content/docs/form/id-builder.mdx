title: ID Builder
---

```ts
type Id = Brand<"sjsf-id">;

interface IdentifiableFieldElement {
  help: {};
  "key-input": {};
  examples: {};
  title: {};
  description: {};
  errors: {};
  oneof: {};
  anyof: {};
  form: {};
  submit: {};
}

type FieldPseudoElement = keyof IdentifiableFieldElement | number;

export interface FormIdBuilder {
  fromPath: (path: Path) => Id;
  pseudoId: (instanceId: Id, element: FieldPseudoElement) => Id;
}

export interface FormIdBuilderToPath {
  toPath(id: Id): Path;
}
```
